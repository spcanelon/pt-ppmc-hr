summary_job_income <- 
  data_all_ee |> 
  mutate(total_employees = n_distinct(ee_id)) |>
  group_by(sub_category) |> 
  mutate(n_job_sub_category = n_distinct(ee_id),
         prop_job_sub_category = n_job_sub_category/total_employees) |> 
  group_by(job_category, sub_category, annual_salary_category, 
           n_job_sub_category, prop_job_sub_category) |> 
  summarize(n_employees = n_distinct(ee_id),
            prop_employees = n_employees/n_job_sub_category) |>
  unique() |> 
  ungroup() |> 
  # convert to uppercase so sorting alphabetically ignores case
  arrange(str_to_upper(job_category)) |> 
  tidyr::pivot_wider(names_from = annual_salary_category,
                     values_from = c(n_employees, prop_employees),
                     names_glue = "{.value}_{annual_salary_category}") |> 
  clean_names()
